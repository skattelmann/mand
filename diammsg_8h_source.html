367  uint32_t code, uint8_t flags, uint32_t vendor_id,
368  time_t data) {
369  uint32_t dtime;
370 
371  if(data + TIME_SINCE_EPOCH > 0xFFFFFFFF)
372  return 1;
373 
374  dtime = htonl((uint32_t)data + TIME_SINCE_EPOCH);
375  return diam_avpgrp_add_raw(ctx, avpgrp, code, flags, vendor_id, &dtime, sizeof(uint32_t));

378  /* accept only timevals with both fields <= 32 bits (standardization) */
379 static inline int
380 diam_avpgrp_add_timeval(void *ctx, DIAM_AVPGRP **avpgrp,
381  uint32_t code, uint8_t flags, uint32_t vendor_id,
382  struct timeval value) {
383  DIAM_TIMEVAL tvalue;
384 
385  if(value.tv_sec > (int32_t)0x7FFFFFFF || value.tv_usec >= 1000000)
386  return 1;
387 
388  tvalue.tv_sec = htonl((uint32_t)value.tv_sec);
389  tvalue.tv_usec = htonl((uint32_t)value.tv_usec);
390 
391  return diam_avpgrp_add_raw(ctx, avpgrp, code, flags, vendor_id, &tvalue, sizeof(DIAM_TIMEVAL));
392 }
