156 static int comp_code(const void *m1, const void *m2)
157 {
158  struct code2str *a = (struct code2str *) m1;
159  struct code2str *b = (struct code2str *) m2;
160 
161  if (a->code > b->code)
162  return 1;
163  else if (a->code < b->code)
164  return -1;
165  else
166  return 0;
167 }

172 const char *_get_cmd(uint16_t code)
173 {
174  struct code2str srch = { .code = code };
175  struct code2str *res;
176 
177  res = bsearch(&srch, &cmd2str, sizeof(cmd2str) / sizeof(struct code2str),
178  sizeof(struct code2str), comp_code);
179  if (res)
180  return res->command;
181 
182  return unk_code;
183 }

207 static void
208 hexdump(void *data, int len) {
209  for(int i = 0; i < len; i++, data++) {
210  if((i % 16) == 0)
211  fprintf(stderr, "\n%08x: ", i);
212 
213  fprintf(stderr, "%02x ", *(uint8_t *)data);
214  }
215  fprintf(stderr, "\n");
216 }

228 static void dump_avpgrp(int level, DIAM_AVPGRP *avpgrp)
229 {
230  uint32_t code;
231  uint8_t flags;
232  uint32_t vendor_id;
233  void *data;
234  size_t len;
235 
236  while(!diam_avpgrp_get_avp(avpgrp, &code, &flags, &vendor_id, &data, &len)) {
237  fprintf(stderr, "%sC: %8d %-20s, F: %02x, V: %8d, L: %6d, %p\n", indent(level), code, _get_avp(code), flags, vendor_id, len, data);
238  if (code == AVP_CONTAINER) {
239  DIAM_AVPGRP *avpgrp1 = diam_decode_avpgrp(NULL, data, len);
240 
241  dump_avpgrp(level + 1, avpgrp1);
242  talloc_free(avpgrp1);
243  }
244  }
245  fprintf(stderr, "\n");
246 }

248 void
249 dump_diam_packet(DIAM_REQUEST *req) {
250  uint32_t code;
251  uint8_t flags;
252  uint32_t vendor_id;
253  void *data;
254  size_t len;
255  DIAM_PACKET *packet = &req->packet;
256 
257  hexdump(packet, diam_packet_length(packet));
258 
259  fprintf(stderr, "Version: %d\n", packet->version);
260  fprintf(stderr, " Length: %d\n", diam_packet_length(packet));
261  fprintf(stderr, " Flags: %02x\n", packet->flags);
262  fprintf(stderr, "Command: %s (%d)\n", _get_cmd(diam_packet_code(packet)), diam_packet_code(packet));
263  fprintf(stderr, " App-Id: %08x\n", ntohl(packet->app_id));
264  fprintf(stderr, " Hop-Id: %08x\n", ntohl(packet->hop2hop_id));
265  fprintf(stderr, " End-Id: %08x\n", ntohl(packet->end2end_id));
266 
267  diam_request_reset_avp(req);
268  while(!diam_request_get_avp(req, &code, &flags, &vendor_id, &data, &len)) {
269  fprintf(stderr, "C: %8d %-20s, F: %02x, V: %8d, L: %6d, %p\n", code, _get_avp(code), flags, vendor_id, len, data);
270  if(code == AVP_CONTAINER) {
271  DIAM_AVPGRP *avpgrp = diam_decode_avpgrp(req, data, len);
272 
273  dump_avpgrp(1, avpgrp);
274  talloc_free(avpgrp);
275  }
276  }
