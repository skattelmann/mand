239 static uint32_t
240 Lua_decode_list(lua_State *L, DIAM_AVPGRP *grp, int *nodes)
241 {
242  uint32_t code;
243  uint8_t flags;
244  uint32_t vendor_id;
245  void *data;
246  size_t len;
247 
248  uint32_t rc;
249  int i;
250 
251  lua_newtable(L);
252 
253  for (i = 1;
254  !diam_avpgrp_get_avp(grp, &code, &flags, &vendor_id, &data, &len);
255  i++) {
256  DIAM_AVPGRP *node_container;
257  uint32_t type;
258 
259  if (code != AVP_CONTAINER || !len)
260  return RC_ERR_MISC;
261  lua_pushinteger(L, i);
262 
263  if (!(node_container = diam_decode_avpgrp(NULL, data, len)))
264  return RC_ERR_ALLOC;
265  lua_newtable(L);
266 
267  if (diam_avpgrp_get_avp(node_container, &code, &flags, &vendor_id,
268  &data, &len) ||
269  code != AVP_NODE_NAME || !len) {
270  dm_grp_free(node_container);
271  return RC_ERR_MISC;
272  }
273  lua_pushlstring(L, data, len);
274  lua_setfield(L, -2, "name");
275 
276  if (diam_avpgrp_get_avp(node_container, &code, &flags, &vendor_id,
277  &data, &len) ||
278  code != AVP_NODE_TYPE || len != sizeof(uint32_t)) {
279  dm_grp_free(node_container);
280  return RC_ERR_MISC;
281  }
282  type = diam_get_uint32_avp(data);
283  lua_pushinteger(L, type);
284  lua_setfield(L, -2, "type");
285 
286  switch (type) {
287  case NODE_PARAMETER:
288  if (diam_avpgrp_get_avp(node_container, &code, &flags,
289  &vendor_id, &data, &len)) {
290  dm_grp_free(node_container);
291  return RC_ERR_MISC;
292  }
293 
294  if (code == AVP_NODE_DATATYPE) {
295  if (len != sizeof(uint32_t)) {
296  dm_grp_free(node_container);
297  return RC_ERR_MISC;
298  }
299 
300  lua_pushinteger(L, diam_get_uint32_avp(data));
301  lua_setfield(L, -2, "datatype");
302  } else {
303  lua_pushinteger(L, code);
304  lua_setfield(L, -2, "datatype");
305 
306  if ((rc = Lua_decode_unknown(L, code, data, len))) {
307  dm_grp_free(node_container);
308  return rc;
309  }
310  lua_setfield(L, -2, "value");
311  }
312 
313  break;
314 
315  case NODE_TABLE:
316  case NODE_OBJECT: {
317  DIAM_AVPGRP *child_grp;
318  int child_nodes;
319 
320  if (type == NODE_TABLE) {
321  if (diam_avpgrp_get_avp(node_container, &code, &flags,
322  &vendor_id, &data, &len))
323  break;
324  } else {
325  if (diam_avpgrp_get_avp(node_container, &code, &flags,
326  &vendor_id, &data, &len)) {
327  dm_grp_free(node_container);
328  return RC_ERR_MISC;
329  }
330 
331  if (code == AVP_NODE_SIZE) {
332  if (len != sizeof(uint32_t)) {
333  dm_grp_free(node_container);
334  return RC_ERR_MISC;
335  }
336 
337  lua_pushinteger(L, diam_get_uint32_avp(data));
338  lua_setfield(L, -2, "size");
339 
340  break;
341  }
342  }
343 
344  if (code != AVP_CONTAINER) {
345  dm_grp_free(node_container);
346  return RC_ERR_MISC;
347  }
348 
349  if (!(child_grp = diam_decode_avpgrp(node_container, data, len))) {
350  dm_grp_free(node_container);
351  return RC_ERR_ALLOC;
352  }
353 
354  if ((rc = Lua_decode_list(L, child_grp, &child_nodes))) {
355  dm_grp_free(node_container);
356  return rc;
357  }
358  lua_setfield(L, -2, "children");
359 
360  lua_pushinteger(L, child_nodes);
361  lua_setfield(L, -2, "size");
362  }
363  }
364 
365  lua_settable(L, -3);
366  dm_grp_free(node_container);
367  }
368 
369  if (nodes)
370  *nodes = i - 1;
371 
372  return RC_OK;
373 }

415  uint32_t type, vendor_id;
416  uint8_t flags;

421  return rc;

423  lua_createtable(L, 0, 3);
424 
425  lua_pushinteger(L, type);
426  lua_setfield(L, -2, "type");
427  lua_pushstring(L, path);
428  lua_setfield(L, -2, "path");
429  free(path);
430 
431  if (diam_avpgrp_get_avp(event, &type, &flags,
432  &vendor_id, &data, &len))
433  return RC_ERR_MISC;
434  if ((rc = Lua_decode_unknown(L, type, data, len)))
435  return rc;
436  lua_setfield(L, -2, "value");
437  break;
438  }
439  case NOTIFY_INSTANCE_CREATED:
440  if ((rc = dm_decode_instance_created(event, &path)))
441  return rc;
442 
443  lua_pushstring(L, path);
444  free(path);
445  break;
446 
447  case NOTIFY_INSTANCE_DELETED:
448  if ((rc = dm_decode_instance_deleted(event, &path)))
449  return rc;
450 
451  lua_pushstring(L, path);
452  free(path);
453  break;
454 
455  default:
456  return RC_ERR_MISC;
457  }
458 
459  lua_setfield(L, -2, "info");
460 
461  lua_settable(L, -3);
462  dm_grp_free(event);
463  }
464 
465  return rc;
466 }

468 static void
469 generic_Lua_callback(DMCONFIG_EVENT event, DMCONTEXT *dmCtx, void *user_data,
470  uint32_t answer_rc, DIAM_AVPGRP *answer_grp)
471 {
472  LUA_CALLBACK *cb = user_data;
473  lua_State *L = cb->L;
474 
475  int top;
476 
477  lua_rawgeti(L, LUA_REGISTRYINDEX, cb->callback_ref);
478  top = lua_gettop(L);
479  lua_pushnumber(L, event);
480  lua_rawgeti(L, LUA_REGISTRYINDEX, cb->context_ref);
481  lua_rawgeti(L, LUA_REGISTRYINDEX, cb->user_data_ref);
482  lua_pushnumber(L, answer_rc);
483 
484  if (event == DMCONFIG_ANSWER_READY && !answer_rc)
485  switch (cb->code) {
486  case CMD_STARTSESSION:
487  if (dm_decode_start_session(dmCtx, answer_grp))
488  L_ERROR(L_ALLOC);
489  break;
490  case CMD_ENDSESSION:
491  dm_context_set_sessionid(dmCtx, 0);
492  break;
493  case CMD_SESSIONINFO:
494  if (Lua_decode_get_session_info(L, answer_grp))
495  L_ERROR(L_MISC);
496  break;
497  case CMD_CFGSESSIONINFO: {
498  uint32_t sessionid, flags;
499  struct timeval timeout;
500 
501  if (dm_decode_get_cfg_session_info(answer_grp, &sessionid, &flags, &timeout))
502  L_ERROR(L_ALLOC);
503 
504  Lua_decode_get_cfg_session_info(L, sessionid, flags, timeout);
505  break;
506  }
507  case CMD_DB_FINDINSTANCE:
508  case CMD_DB_ADDINSTANCE: {
509  uint16_t instance;
510 
511  if (dm_decode_uint16(answer_grp, &instance))
512  L_ERROR(L_ALLOC);
513  lua_pushnumber(L, instance);
514  break;
515  }
516  case CMD_DB_DUMP: {
517  char *data;
518 
519  if (dm_decode_string(answer_grp, &data))
520  L_ERROR(L_ALLOC);
521  lua_pushstring(L, data);
522  free(data);
523  break;
524  }
525  case CMD_DB_GET:
526  if (Lua_decode_get(L, answer_grp))
527  L_ERROR(L_ALLOC);
528  break;
529  case CMD_DB_LIST:
530  if (Lua_decode_list(L, answer_grp, NULL))
531  L_ERROR(L_ALLOC);
532  break;
533  case CMD_DB_RETRIEVE_ENUMS:
534  if (Lua_decode_retrieve_enums(L, answer_grp))
535  L_ERROR(L_ALLOC);
536  break;
537 
538  case CMD_GET_PASSIVE_NOTIFICATIONS:
539  if (Lua_decode_notifications(L, answer_grp))
540  L_ERROR(L_ALLOC);
541  break;
542  }
543 
544  lua_call(L, lua_gettop(L) - top, 0);
545 
546  luaL_unref(L, LUA_REGISTRYINDEX, cb->context_ref);
547  luaL_unref(L, LUA_REGISTRYINDEX, cb->callback_ref);
548  luaL_unref(L, LUA_REGISTRYINDEX, cb->user_data_ref);
549 
550  cb->callbackDone = 1; /* req. will be removed from the request list */
551 }
