124 uint32_t
125 event_aux_dmRead(int fd, short event, COMMCONTEXT *readCtx,
126  uint8_t *alreadyRead, COMMSTATUS *status)
127 {
128  ssize_t length;
129  uint32_t bufsize;
130  uint32_t len;
131 
132  if (event == EV_TIMEOUT)
133  aux_RET(CONNRESET, RC_ERR_CONNECTION);
134 
135  if (event != EV_READ) /* a number of events can be ignored */
136  aux_RET(INCOMPLETE, RC_OK);
137 
138  /* nothing or not enough read -> read as much as possible */
139 
140  if (!*alreadyRead) {
141  *alreadyRead = 1;
142 
143  bufsize = readCtx->cAlloc * BUFFER_CHUNK_SIZE;
144  do {
145  /* don't reallocate if there's still space */
146  if (readCtx->bytes == bufsize) {
147  readCtx->cAlloc++;
148  bufsize += BUFFER_CHUNK_SIZE;
149 
150  /* allocate a DM_REQUEST structure (we're reading into the REQUEST structure) */
151  if (!(readCtx->req = talloc_realloc_size(NULL, readCtx->req,
152  sizeof(DM_REQUEST_INFO) + bufsize)))
153  aux_RET(ERROR, RC_ERR_ALLOC);
154 
155  readCtx->buffer = (uint8_t*)readCtx->req +
156  sizeof(DM_REQUEST_INFO);
157  }
158 
159  do {
160  if ((length = read(fd, readCtx->buffer + readCtx->bytes,
161  bufsize - readCtx->bytes)) == -1) {
162  debug(": read error: %d (%m)", errno);
163  switch (errno) {
164  case EWOULDBLOCK: /* happens if data to read is multiple of BUFFER_CHUNK_SIZE */
165  length = 0;
166  break;
167  case EINTR:
168  break;
169  case ETIMEDOUT:
170  case ECONNRESET:
171  aux_RET(CONNRESET,
172  RC_ERR_CONNECTION);
173  default:
174  aux_RET(ERROR, RC_ERR_CONNECTION);
175  }
176  }
177  else if (!length)
178  aux_RET(CONNRESET, RC_ERR_CONNECTION);
179  /* if length is nevertheless 0, there was an EWOULDBLOCK */
180  } while (length == -1); /* errno is EINTR */
181 
182  readCtx->bytes += length;
183  debug(": read length: %d, total: %d (from %d)", (int)length, readCtx->bytes, bufsize);
184  } while (length && readCtx->bytes == bufsize);
185  } else if (readCtx->bytes >= 4 &&
186  readCtx->bytes >=
187  (len = dm_packet_length(&readCtx->req->packet))) {
188 
189  debug(": read continue, got total: %d, req: %d", readCtx->bytes, len);
190 
191  /* foremost request was also the last one */
192  if (!(readCtx->bytes - len)) {
193  debug(": remove request");
194  talloc_free(readCtx->req);
195  readCtx->req = NULL;
196  readCtx->cAlloc = readCtx->bytes = 0;
197  aux_RET(NOTHING, RC_OK);
198  }
199 
200  /* remove request */
201  readCtx->bytes -= len;
202  memmove(readCtx->buffer, readCtx->buffer + len, readCtx->bytes);
203  }
204 
205  /* process read data */
206 
207  /* less than a part of one request's header to process */
208  if (readCtx->bytes < 4 ||
209  readCtx->bytes < dm_packet_length(&readCtx->req->packet))
210  aux_RET(INCOMPLETE, RC_OK);
211 
212  postprocessRequest(readCtx); /* one or more requests to process */
213  aux_RET(COMPLETE, RC_OK);
214 }

217 uint32_t
218 event_aux_dmWrite(int fd, short event, COMMCONTEXT *writeCtx,
219  COMMSTATUS *status)
220 {
221  ssize_t length;
222 
223  struct sigaction action, oldaction;
224 
225  if (event == EV_TIMEOUT)
226  aux_RET(CONNRESET, RC_ERR_CONNECTION);
227 
228  if (event != EV_WRITE)
229  aux_RET(INCOMPLETE, RC_OK);
230 
231  if (!writeCtx->buffer) {
232  writeCtx->buffer = (uint8_t *)&writeCtx->req->packet;
233  writeCtx->bytes = writeCtx->req->info.size -
234  sizeof(DM_REQUEST_INFO);
235  }
236 
237  memset(&action, 0, sizeof(struct sigaction));
238  action.sa_handler = SIG_IGN;
239  sigaction(SIGPIPE, &action, &oldaction);
240 
241  do {
242  if ((length = write(fd, writeCtx->buffer,
243  writeCtx->bytes)) == -1)
244  switch (errno) {
245  case EAGAIN: /* can only happen when it tries to write the second request in one write event */
246  aux_RET_SIG(NOTHING, RC_OK);
247  case EPIPE:
248  case ECONNRESET:
249  aux_RET_SIG(CONNRESET, RC_ERR_CONNECTION);
250  case EINTR:
251  break;
252  default:
253  aux_RET_SIG(ERROR, RC_ERR_CONNECTION);
254  }
255  else if (!length)
256  aux_RET_SIG(CONNRESET, RC_ERR_CONNECTION);
257  } while (length == -1); /* errno is EINTR */
258 
259  writeCtx->buffer += length;
260  writeCtx->bytes -= length;
261 
262  aux_RET_SIG((writeCtx->bytes ? INCOMPLETE : COMPLETE), RC_OK);
263 }

303 uint32_t
304 dm_register_connect_callback(DMCONTEXT *dmCtx, int type,
305  DMCONFIG_CONNECT_CALLBACK callback, void *userdata)
306 {
307  CONNEVENTCTX *ctx;
308  struct timeval timeout;
309 
310  int fd = dm_context_get_socket(dmCtx);
311  int flags;
312 
313  struct sockaddr_un sockaddr_un;
314  struct sockaddr_in sockaddr_in;
315 
316  struct sockaddr *sockaddr;
317  socklen_t sockaddr_len;
318 
319  if ((flags = fcntl(fd, F_GETFL)) == -1 ||
320  fcntl(fd, F_SETFL, flags | O_NONBLOCK))
321  return RC_ERR_CONNECTION;
322 
323  if (type == AF_UNIX) {
324  memset(&sockaddr_un, 0, sizeof(sockaddr_un));
325 
326  sockaddr_un.sun_family = AF_UNIX;
327  strncpy(sockaddr_un.sun_path + 1, SERVER_LOCAL,
328  sizeof(sockaddr_un.sun_path) - 1);
329 
330  sockaddr = (struct sockaddr*)&sockaddr_un;
331  sockaddr_len = sizeof(sockaddr_un);
332  } else { /* AF_INET */
333  memset(&sockaddr_in, 0, sizeof(sockaddr_in));
334 
335  sockaddr_in.sin_family = AF_INET;
336  sockaddr_in.sin_port = htons(SERVER_PORT);
337  sockaddr_in.sin_addr.s_addr = htonl(SERVER_IP);
338 
339  sockaddr = (struct sockaddr*)&sockaddr_in;
340  sockaddr_len = sizeof(sockaddr_in);
341  }
342 
343  while (connect(fd, sockaddr, sockaddr_len) == -1)
344  if (errno == EINPROGRESS)
345  break;
346  else if (errno == EAGAIN)
347  continue;
348  else
349  return RC_ERR_CONNECTION;
350 
351  if (!(ctx = talloc(NULL, CONNEVENTCTX)))
352  return RC_ERR_ALLOC;
353 
354  ctx->callback = callback;
355  ctx->user_data = userdata;
356  ctx->dmCtx = dmCtx;
357 
358  event_set(&ctx->event, fd, EV_WRITE, connectEvent, ctx);
359  event_base_set(dm_context_get_event_base(dmCtx), &ctx->event);
360 
361  timeout.tv_sec = TIMEOUT_WRITE_REQUESTS; /* if the connect was already successful, there should be no delay */
362  timeout.tv_usec = 0;
363 
364  if (event_add(&ctx->event, &timeout)) {
365  talloc_free(ctx);
366  return RC_ERR_ALLOC;
367  }
368 
369  return RC_OK;
370 }

388 uint32_t
389 dm_generic_register_request(DMCONTEXT *dmCtx, uint32_t code, DM_AVPGRP *grp,
390  DMCONFIG_CALLBACK callback, void *callback_ud)
391 {
392  DM_AVPGRP *completegrp;
393 
394  REQUESTINFO *new, *cur;
395 
396  struct timeval timeout;
397 
398  switch (hopid) { /* one never knows... */
399  case 0: srand((unsigned int)time(NULL));
400  case MAX_INT: hopid = endid = (float)rand()/RAND_MAX * (MAX_INT-1) + 1;
401  break;
402  default: hopid = ++endid;
403  }
404 
405  if (!dmCtx->requestlist_head) {
406  if (!(dmCtx->requestlist_head = talloc(NULL, REQUESTINFO)))
407  return RC_ERR_ALLOC;
408  memset(dmCtx->requestlist_head, 0, sizeof(REQUESTINFO));
409  }
410  if (!(new = talloc(dmCtx->requestlist_head, REQUESTINFO))) {
411  talloc_free(dmCtx->requestlist_head);
412  return RC_ERR_ALLOC;
413  }
414 
415  if (!(new->request = new_dm_request(new, code, CMD_FLAG_REQUEST,
416  APP_ID, hopid, endid))) {
417  talloc_free(dmCtx->requestlist_head);
418  return RC_ERR_ALLOC;
419  }
420 
421  if (!(completegrp = new_dm_avpgrp(new->request)) ||
422  dm_avpgrp_add_uint32(new->request, &completegrp, AVP_SESSIONID, 0,
423  VP_TRAVELPING, dmCtx->sessionid) ||
424  (grp && dm_avpgrp_add_avpgrp(new->request, &completegrp,
425  AVP_CONTAINER, 0, VP_TRAVELPING,
426  grp)) ||
427  build_dm_request(new, &new->request, completegrp)) {
428  talloc_free(dmCtx->requestlist_head);
429  return RC_ERR_ALLOC;
430  }
431  talloc_free(completegrp);
432 
433 #ifdef LIBDMCONFIG_DEBUG
434  if (dmconfig_debug_level) {
435  fprintf(stderr, "Send request:\n");
436  dump_dm_packet(new->request);
437  }
438 #endif
439 
440  new->callback = callback;
441  new->user_data = callback_ud;
442  new->dmCtx = dmCtx;
443  new->status = REQUEST_SHALL_WRITE;
444  new->code = code;
445  new->hopid = hopid;
446  new->next = NULL;
447 
448  for (cur = dmCtx->requestlist_head; cur->next; cur = cur->next);
449  cur->next = new;
450 
451  if (!event_pending(&dmCtx->writeCtx.event, EV_WRITE | EV_PERSIST, NULL)) {
452  event_set(&dmCtx->writeCtx.event, dmCtx->socket,
453  EV_WRITE | EV_PERSIST, writeEvent, dmCtx);
454  event_base_set(dm_context_get_event_base(dmCtx), &dmCtx->writeCtx.event);
455 
456  timeout.tv_sec = TIMEOUT_WRITE_REQUESTS;
457  timeout.tv_usec = 0;
458 
459  if (event_add(&dmCtx->writeCtx.event, &timeout)) {
460  talloc_free(dmCtx->requestlist_head);
461  return RC_ERR_ALLOC;
462  }
463  }
464 
465  return RC_OK;
466 }

1019 static void
1020 writeEvent(int fd, short event, void *arg)
1021 {
1022  DMCONTEXT *dmCtx = arg;
1023  COMMCONTEXT *ctx = &dmCtx->writeCtx;
1024 
1025  COMMSTATUS status;
1026  uint32_t rc;
1027 
1028  struct timeval timeout;
1029 
1030  do {
1031  if (!ctx->buffer) {
1032  for (ctx->cur_request = dmCtx->requestlist_head->next;
1033  ctx->cur_request &&
1034  ctx->cur_request->status != REQUEST_SHALL_WRITE;
1035  ctx->cur_request = ctx->cur_request->next);
1036  if (!ctx->cur_request) { /* all requests written */
1037  event_del(&ctx->event);
1038  return;
1039  }
1040  ctx->cur_request->status = REQUEST_WRITING;
1041  ctx->req = ctx->cur_request->request;
1042  }
1043 
1044  rc = event_aux_dmWrite(fd, event, ctx, &status);
1045  switch (status) {
1046  case COMPLETE:
1047  talloc_free(ctx->cur_request->request);
1048  ctx->buffer = NULL;
1049 
1050  ctx->cur_request->status = REQUEST_SHALL_READ;
1051 
1052  if (!event_pending(&dmCtx->readCtx.event,
1053  EV_READ | EV_PERSIST, NULL)) {
1054  event_set(&dmCtx->readCtx.event, dmCtx->socket,
1055  EV_READ | EV_PERSIST, readEvent, dmCtx);
1056  event_base_set(dm_context_get_event_base(dmCtx),
1057  &dmCtx->readCtx.event);
1058 
1059  timeout.tv_sec = TIMEOUT_READ_REQUESTS;
1060  timeout.tv_usec = 0;
1061 
1062  if (event_add(&dmCtx->readCtx.event, &timeout)) {
1063  rc = RC_ERR_ALLOC;
1064  break;
1065  }
1066  }
1067 
1068  timeout.tv_sec = TIMEOUT_WRITE_REQUESTS;
1069  timeout.tv_usec = 0;
1070 
1071  if (event_add(&ctx->event, &timeout)) /* increase writeEvent's timeout */
1072  rc = RC_ERR_ALLOC;
1073 
1074  break;
1075  case INCOMPLETE:
1076  timeout.tv_sec = TIMEOUT_CHUNKS;
1077  timeout.tv_usec = 0;
1078 
1079  if (event_add(&ctx->event, &timeout)) { /* reduce writeEvent's timeout */
1080  rc = RC_ERR_ALLOC;
1081  break;
1082  }
1083  case NOTHING:
1084  return;
1085  default: /* CONNRESET or ERROR */
1086  break;
1087  }
1088  } while (!rc);
1089 
1090  CALLBACK(ctx->cur_request, DMCONFIG_ERROR_WRITING,
1091  ctx->cur_request->dmCtx, ctx->cur_request->user_data, rc, NULL);
1092 
1093  L_FOREACH(REQUESTINFO, cur, dmCtx->requestlist_head)
1094  if (cur->status != REQUEST_SHALL_READ)
1095  talloc_free(cur->request);
1096  talloc_free(dmCtx->requestlist_head);
1097  dmCtx->requestlist_head = NULL;
1098 
1099  ctx->cur_request = NULL;
1100  ctx->req = NULL;
1101  ctx->buffer = NULL;
1102  event_del(&ctx->event);
1103 
1104  talloc_free(dmCtx->readCtx.req);
1105  dmCtx->readCtx.req = NULL;
1106 
1107  if (event_pending(&dmCtx->readCtx.event, EV_READ | EV_PERSIST, NULL))
1108  event_del(&dmCtx->readCtx.event);
1109 }

1113 static void
1114 readEvent(int fd, short event, void *arg)
1115 {
1116  DMCONTEXT *dmCtx = arg;
1117  COMMCONTEXT *ctx = &dmCtx->readCtx;
1118 
1119  uint32_t avpcode;
1120  uint8_t flags;
1121  uint32_t vendor_id;
1122  void *data;
1123  size_t len;
1124 
1125  uint8_t alreadyRead = 0;
1126 
1127  struct timeval timeout;
1128 
1129  for (;;) {
1130  uint32_t hopid;
1131  REQUESTINFO *cur, *reqEl;
1132 
1133  COMMSTATUS status;
1134  uint32_t rc;
1135 
1136  rc = event_aux_dmRead(fd, event, ctx, &alreadyRead, &status);
1137  switch (status) {
1138  case INCOMPLETE:
1139  timeout.tv_sec = TIMEOUT_CHUNKS;
1140  timeout.tv_usec = 0;
1141 
1142  if (event_add(&ctx->event, &timeout)) /* reduce readEvent's timeout */
1143  goto abort;
1144  case NOTHING:
1145  return;
1146  case COMPLETE:
1147  break;
1148  default: /* CONNRESET or ERROR */
1149  goto abort;
1150  }
1151 
1152 #ifdef LIBDMCONFIG_DEBUG
1153  if (dmconfig_debug_level) {
1154  fprintf(stderr, "Recieved %s:\n",
1155  dm_packet_flags(&ctx->req->packet) &
1156  CMD_FLAG_REQUEST ? "request" : "answer");
1157  dump_dm_packet(ctx->req);
1158  dm_request_reset_avp(ctx->req);
1159  }
1160 #endif
1161 
1162  /* server request */
1163  if (dm_packet_flags(&ctx->req->packet) & CMD_FLAG_REQUEST) {
1164  switch (dm_packet_code(&ctx->req->packet)) {
1165 
1166  default:
1167  goto abort;
1168  }
1169  } else {
1170  if (!dmCtx->requestlist_head || !dmCtx->requestlist_head->next)
1171  goto abort;
1172 
1173  hopid = dm_hop2hop_id(&ctx->req->packet);
1174 
1175  for (cur = dmCtx->requestlist_head;
1176  cur->next && cur->next->hopid != hopid;
1177  cur = cur->next);
1178  if (!cur->next || cur->next->status != REQUEST_SHALL_READ)
1179  goto abort;
1180  reqEl = cur->next;
1181  cur->next = reqEl->next;
1182 
1183  if (dm_request_get_avp(ctx->req, &avpcode, &flags,
1184  &vendor_id, &data, &len) ||
1185  avpcode != AVP_RC || len != sizeof(uint32_t)) {
1186  CALLBACK(reqEl, DMCONFIG_ERROR_READING, reqEl->dmCtx, reqEl->user_data, RC_ERR_MISC, NULL);
1187  goto cleanup;
1188  }
1189 
1190  if ((rc = dm_get_uint32_avp(data))) {
1191  CALLBACK(reqEl, DMCONFIG_ANSWER_READY, reqEl->dmCtx, reqEl->user_data, rc, NULL);
1192 
1193  /* clean up callback structures if necessary, so the read event is deleted */
1194  switch (reqEl->code) {
1195  case CMD_SUBSCRIBE_NOTIFY:
1196  memset(&dmCtx->callbacks.active_notification, 0,
1197  sizeof(ACTIVE_NOTIFY_INFO));
1198  break;
1199  }
1200 
1201  goto cleanup;
1202  } else /* RC_OK */ {
1203  switch (reqEl->code) {
1204  case CMD_UNSUBSCRIBE_NOTIFY:
1205  memset(&dmCtx->callbacks.active_notification, 0,
1206  sizeof(ACTIVE_NOTIFY_INFO));
1207  break;
1208 
1209  case CMD_ENDSESSION: /*
1210  * allows the implicit abortion (deletion of read event -> event loop returns) of
1211  * asynchronous processes (ping, active notify, etc)
1212  */
1213  memset(&dmCtx->callbacks, 0,
1214  sizeof(struct _dmContext_callbacks));
1215  break;
1216  }
1217  }
1218 
1219  if (dm_request_get_avp(ctx->req, &avpcode, &flags, &vendor_id, &data, &len)) {
1220  CALLBACK(reqEl, DMCONFIG_ANSWER_READY, reqEl->dmCtx, reqEl->user_data, RC_OK, NULL);
1221  } else if (avpcode == AVP_CONTAINER) {
1222  DM_AVPGRP *answer;
1223 
1224  if ((answer = dm_decode_avpgrp(ctx->req, data, len))) {
1225  CALLBACK(reqEl, DMCONFIG_ANSWER_READY, reqEl->dmCtx, reqEl->user_data, RC_OK, answer);
1226  } else {
1227  CALLBACK(reqEl, DMCONFIG_ERROR_READING, reqEl->dmCtx, reqEl->user_data, RC_ERR_ALLOC, NULL);
1228  }
1229  } else {
1230  CALLBACK(reqEl, DMCONFIG_ERROR_READING, reqEl->dmCtx, reqEl->user_data, RC_ERR_MISC, NULL);
1231  }
1232 
1233 cleanup:
1234 
1235  if (dmCtx->requestlist_head->next)
1236  talloc_free(reqEl);
1237  else {
1238  talloc_free(dmCtx->requestlist_head);
1239  dmCtx->requestlist_head = NULL;
1240  }
1241  }
1242 
1243  for (cur = dmCtx->requestlist_head;
1244  cur && cur->status != REQUEST_SHALL_READ;
1245  cur = cur->next);
1246  if (!cur && /* nothing more to read (at least not expected) */
1247  !dmCtx->callbacks.active_notification.callback) { /* FIXME: a reference counter would be cleaner */
1248  event_del(&ctx->event);
1249  /* if there's nothing more to read, the fields are reset and NOTHING is returned */
1250  event_aux_dmRead(fd, EV_READ, ctx,
1251  &alreadyRead, &status);
1252  if (status != NOTHING)
1253  goto abort;
1254 
1255  return;
1256  }
1257 
1258  timeout.tv_sec = TIMEOUT_READ_REQUESTS;
1259  timeout.tv_usec = 0;
1260 
1261  if (event_add(&ctx->event, &timeout)) /* increase readEvent's timeout */
1262  break;
1263  }
1264 
1265 abort:
1266 
1267  event_del(&ctx->event);
1268  talloc_free(ctx->req);
1269  ctx->req = NULL;
1270  ctx->cAlloc = ctx->bytes = 0;
1271  memset(&dmCtx->callbacks, 0, sizeof(struct _dmContext_callbacks));
1272 }

1611  ? RC_ERR_ALLOC : RC_OK;
1612  case AVP_ENUM:
1613  case AVP_PATH:
