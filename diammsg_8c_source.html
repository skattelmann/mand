30 int
31 get_avp(uint32_t *code, uint8_t *flags, uint32_t *vendor_id,
32  void **data, size_t *len, DIAM_AVP **avp) {
33  int avp_len;
34 
35  if((avp_len = diam_avp_length(*avp)) < 8)
36  return DIAM_INVALID_AVP_LENGTH;
37 
38  if((*avp)->flags & AVP_FLAG_RESERVED)
39  return DIAM_INVALID_AVP_BITS;
40 
41  *code = ntohl((*avp)->code);
42  *flags = (*avp)->flags & (AVP_FLAG_MANDATORY | AVP_FLAG_PRIVAT);
43  *len = avp_len - 8;
44  *data = (uint8_t*)*avp + 8;
45 
46  if((*avp)->flags & AVP_FLAG_VENDOR) {
47  if(avp_len < 12)
48  return DIAM_INVALID_AVP_LENGTH;
49  *vendor_id = ntohl((*avp)->vendor_id);
50  *len -= 4;
51  *data += 4;
52  }
53  *(uint8_t**)avp += PAD32(avp_len);
54 
55  return 0;
56 }

91  size_t pkt_len;
92  DIAM_PACKET *pkt = (DIAM_PACKET *)data;
93 
94  assert(req != NULL);
95  if (len < 4)
96  return 0;
97 
98  pkt_len = diam_packet_length(pkt);
99  if (len > pkt_len)
100  len = pkt_len;
101 
102  if(!(*req = talloc_size(ctx, sizeof(DIAM_REQUEST_INFO) + diam_packet_length(pkt))))
103  return 0;
104 
105  (*req)->info.size = sizeof(DIAM_REQUEST_INFO) + diam_packet_length(pkt);
106  (*req)->info.avpptr = (DIAM_AVP*)((uint8_t*)(*req) + sizeof(DIAM_REQUEST) + len);
107  memcpy(&(*req)->packet, data, len);
108 
109  return len;
110 }

220 int
221 diam_avpgrp_add_raw(void *ctx, DIAM_AVPGRP **avpgrp,
222  uint32_t code, uint8_t flags, uint32_t vendor_id,
223  const void *data, size_t len) {
224  DIAM_AVP *avp;
225  uint8_t *payload;
226  int avp_len;
227  uint32_t new_size;
228 
229  if((new_size = sizeof(DIAM_AVPGRP) + (*avpgrp)->info.avps_length +
230  sizeof(DIAM_AVP) + PAD32(len)) > (*avpgrp)->info.size) {
231  (*avpgrp)->info.size = (new_size/DIAM_BLOCK_ALLOC + 1) * DIAM_BLOCK_ALLOC;
232  if(!(*avpgrp = talloc_realloc_size(ctx, *avpgrp, (*avpgrp)->info.size)))
233  return 1;
234  }
235 
236  avp = (DIAM_AVP*)((uint8_t*)*avpgrp + sizeof(DIAM_AVPGRP) + (*avpgrp)->info.avps_length);
237  payload = (uint8_t*) avp + 8;
238  avp_len = len + 8;
239 
240  avp->code = htonl(code);
241  avp->flags = flags & (AVP_FLAG_MANDATORY | AVP_FLAG_PRIVAT);
242  if(vendor_id) {
243  avp->flags |= AVP_FLAG_VENDOR;
244  avp->vendor_id = htonl(vendor_id);
245  payload += 4;
246  avp_len += 4;
247  }
248 
249  uint32to24(avp->length, avp_len);
250  memcpy(payload, data, len);
251 
252  memset(payload + len, 0, PAD32(avp_len) - avp_len);
253  (*avpgrp)->info.avps_length += PAD32(avp_len);
254 
255  return 0;
256 }

260  uint32_t code, uint8_t flags, uint32_t vendor_id,
261  int af, const void *data) {

266  struct in6_addr in6;
267  };
268  } addr;
269

271  addr.af = htons(IANA_INET);
272  addr.in = *(struct in_addr *)data;
273  return diam_avpgrp_add_raw(ctx, avpgrp, code, flags, vendor_id, &addr, sizeof(struct in_addr) + 2);
274  } else if(af == AF_INET6) {
275  addr.af = htons(IANA_INET6);
276  addr.in6 = *(struct in6_addr *)data;
277  return diam_avpgrp_add_raw(ctx, avpgrp, code, flags, vendor_id, &addr, sizeof(struct in6_addr) + 2);
278  } else
